# Convert dictionaries to lists for plotting
labels = ['different avatar','same avatar']

neutral_means = [ plot_means['choice_diff_neutral'],plot_means['choice_same_neutral']]
gender_means = [ plot_means['choice_diff_gender'], plot_means['choice_same_gender']]
neutral_cis = [ plot_cis['choice_diff_neutral'], plot_cis['choice_same_neutral']]
gender_cis = [plot_cis['choice_diff_gender'], plot_cis['choice_same_gender']]


# Set up the figure and axis
fig, ax = plt.subplots(figsize=(8,4),dpi=300)

# Define bar width and positions
bar_width = 0.35
index = np.arange(len(labels))

# Create bars
bar1 = plt.bar(index, neutral_means, bar_width, yerr=neutral_cis, label='neutral avatar', capsize=5, color='0.85')
bar2 = plt.bar(index + bar_width, gender_means, bar_width, yerr=gender_cis, label='gender avatar', capsize=5, color='0.25')


# Set y-limit
plt.ylim([0, 1])

# Remove top and right spines
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)

ax.yaxis.grid(True, ls=':')

# Add a horizontal line with ticks at both ends and an asterisk on top
y_value_for_line = max(neutral_means[0] + neutral_cis[0], gender_means[0] + gender_cis[0]) + 0.08  # Position above the highest error bar
plt.plot([index[1], index[1] + bar_width], [y_value_for_line, y_value_for_line], color='black')
# Add an asterisk (*) on top of the line
plt.annotate('***', xy=(index[1] + bar_width / 2, y_value_for_line), xytext=(0, 3), textcoords='offset points', ha='center', va='bottom')
# Add short vertical lines as ticks at both ends
tick_length = 0.01  # Length of the tick
plt.plot([index[1], index[1]], [y_value_for_line - tick_length, y_value_for_line + tick_length], color='black')
plt.plot([index[1] + bar_width, index[1] + bar_width], [y_value_for_line - tick_length, y_value_for_line + tick_length], color='black')


plt.plot([index[0], index[0] + bar_width], [y_value_for_line, y_value_for_line], color='black')
# Add an asterisk (*) on top of the line
plt.annotate('**', xy=(index[0] + bar_width / 2, y_value_for_line), xytext=(0, 3), textcoords='offset points', ha='center', va='bottom')
# Add short vertical lines as ticks at both ends
tick_length = 0.01  # Length of the tick
plt.plot([index[0], index[0]], [y_value_for_line - tick_length, y_value_for_line + tick_length], color='black')
plt.plot([index[0] + bar_width, index[0] + bar_width], [y_value_for_line - tick_length, y_value_for_line + tick_length], color='black')

# Add some text for labels, title and axes ticks
ax.set_ylabel('proportion of choosing the risky option')
# plt.title('Grouped bar chart with 95% Confidence Intervals')
plt.xticks(index + bar_width / 2, labels)  # Position the group labels in the center of the grouped bars
plt.legend()

# Show the plot
plt.tight_layout()
plt.show()
